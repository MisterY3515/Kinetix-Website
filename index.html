class Counter {
    state count: int = 0

    fn increment(&mut self) {
        self.count = self.count + 1
    }
}

fn main() {
    let mut c = Counter { count: 0 }
    
    effect [c.count] {
        println("Count changed to: " + c.count)
    }

    c.increment() // Triggers effect implicitly
}
